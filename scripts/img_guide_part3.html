4.2 Kaynak İstemcileri — Stok Fotoğraf

Her ücretsiz kaynak için ortak arayüzlü bir istemci sınıfı. Hepsi search() metodu sunar ve standart bir sonuç sözlüğü döndürür.

clients/unsplash_client.py
import aiohttp
from config import Config

class UnsplashClient:
    BASE = "https://api.unsplash.com"

    def __init__(self, key):
        self.headers = {
            "Authorization": f"Client-ID {key}",
            "Accept-Version": "v1",
        }

    async def search(self, session, query):
        """Unsplash'ta ara, en kaliteli sonucu döndür."""
        params = {
            "query": query, "per_page": 10,
            "orientation": "landscape",
            "content_filter": "high",  # güvenli içerik
        }
        async with session.get(f"{self.BASE}/search/photos",
                params=params, headers=self.headers) as resp:
            if resp.status != 200:
                return None
            data = await resp.json()

        # Skorla: beğeni sayısı + min genişlik 1200
        scored = []
        for r in data.get("results", []):
            if r["width"] < 1200:
                continue
            score = min(r.get("likes", 0) / 200, 1.0)
            scored.append((score, r))
        if not scored:
            return None
        scored.sort(key=lambda x: -x[0])
        score, best = scored[0]
        if best.get("likes", 0) < Config.MIN_UNSPLASH_LIKES:
            return None

        # ZORUNLU: Unsplash atıf kuralı — download tetikle
        try:
            async with session.get(best["links"]["download_location"],
                    headers=self.headers):
                pass
        except Exception:
            pass

        return {
            "url": best["urls"]["full"],
            "source": "unsplash",
            "source_id": best["id"],
            "photographer": best["user"]["name"],
            "photographer_url": best["user"]["links"]["html"],
            "license": "unsplash", "score": score,
        }

from config import Config

class PexelsClient:
    BASE = "https://api.pexels.com/v1"

    def __init__(self, key):
        self.headers = {"Authorization": key}

    async def search(self, session, query):
        params = {"query": query, "per_page": 10,
                  "orientation": "landscape", "size": "large"}
        async with session.get(f"{self.BASE}/search",
                params=params, headers=self.headers) as resp:
            if resp.status != 200:
                return None
            data = await resp.json()

        if not photos:

        if best["width"] < 1200:
            return None
        return {

            "source": "pexels", "source_id": str(best["id"]),
            "photographer": best["photographer"],
            "photographer_url": best["photographer_url"],
            "license": "pexels", "score": 0.75,

import aiohttp
from config import Config

class PixabayClient:
    BASE = "https://pixabay.com/api/"

    def __init__(self, key):
        self.key = key

    async def search(self, session, query):
        params = {
            "key": self.key, "q": query,
            "image_type": "photo", "orientation": "horizontal",
            "category": "food", "per_page": 10,
            "safesearch": "true", "min_width": 1200,
        }
        async with session.get(self.BASE, params=params) as resp:
            if resp.status != 200:
                return None
            data = await resp.json()
        hits = data.get("hits", [])
        # İndirme sayısı eşiği — kalite göstergesi
        good = [h for h in hits
                if h.get("downloads", 0) >= Config.MIN_PIXABAY_DOWNLOADS]
        if not good:
            return None
        best = max(good, key=lambda h: h["downloads"])
        return {
            "url": best["largeImageURL"],
            "source": "pixabay", "source_id": str(best["id"]),
            "photographer": best.get("user", ""),
            "license": "pixabay", "score": 0.68,
        }

4.3 AI Üretim Modülü — DALL·E 3

Stok kaynaklar başarısız olduğunda devreye girer. Style Bible'a uygun prompt şablonuyla görsel üretir.

clients/ai_generator.py
from openai import AsyncOpenAI
from config import Config

# Style Bible uyumlu prompt şablonu (Bölüm 3.3)
PROMPT_TEMPLATE = (
    "Professional food photography of {dish}, "
    "{context}, top-down 90 degree overhead shot, "
    "natural soft window light from the left, "
    "served on a neutral ceramic plate, "
    "rustic wooden table background, realistic textures, "
    "appetizing, warm tones, minimal styling, "
    "no text, no watermark, not cartoon, "
    "shallow depth of field, high resolution, "
    "food magazine quality"
)
TURKISH_CTX = ("traditional Turkish cuisine presentation, "
               "authentic home-style serving")
GENERIC_CTX = "clean modern food styling"

class AIGenerator:
    def __init__(self, api_key):
        self.client = AsyncOpenAI(api_key=api_key)
        self.calls_made = 0

    async def generate(self, dish_name, cuisine):
        """Tarif için AI görseli üret. Maliyet kontrollü."""
        # Maliyet koruması — run başına limit
        if self.calls_made >= Config.DALLE_MAX_PER_RUN:
            raise RuntimeError("AI çağrı limiti doldu")

        ctx = TURKISH_CTX if cuisine == "turkish" else GENERIC_CTX
        prompt = PROMPT_TEMPLATE.format(dish=dish_name, context=ctx)

        resp = await self.client.images.generate(
            model="dall-e-3",
            prompt=prompt,
            size="1792x1024",    # 4:3'e en yakın yatay
            quality="standard",   # $0.040 (hd: $0.080)
            n=1,
        )
        self.calls_made += 1
        return {
            "url": resp.data[0].url,
            "source": "dalle",
            "source_id": f"dalle-{dish_name[:20]}",
            "license": "ai-generated",
            "score": 0.80,
            "requires_review": True,  # AI görseli → insan onayı
        }
💰 Maliyet Koruması
DALLE_MAX_PER_RUN sınırı, tek bir hatalı çalıştırmanın yüzlerce dolarlık fatura çıkarmasını engeller. Her AI çağrısı sayılır; limit dolunca pipeline durur ve kalan işler bir sonraki çalıştırmaya bırakılır.

4.4 Standardizasyon ve İşleme

Kaynaktan gelen ham görsel, Style Bible'a uygun hale getirilir: 1200×900 kırpma, WebP'ye çevirme, thumbnail ve BlurHash üretimi.

processing/standardize.py
from io import BytesIO
from PIL import Image
import blurhash
from config import Config

def standardize(raw_bytes, recipe_id, out_dir):
    """Ham görseli 3 standart çıktıya dönüştür."""
    img = Image.open(BytesIO(raw_bytes)).convert("RGB")

    # 1) Merkeze göre 4:3 kırp
    img = crop_to_ratio(img, 4, 3)

    # 2) FULL — 1200x900 WebP q85
    full = img.resize((Config.FULL_WIDTH, Config.FULL_HEIGHT),
                       Image.LANCZOS)
    full.save(f"{out_dir}/{recipe_id}_full.webp",
              "WEBP", quality=Config.WEBP_QUALITY_FULL)

    # 3) THUMB — 400x300 WebP q80
    thumb = img.resize((Config.THUMB_WIDTH, Config.THUMB_HEIGHT),
                        Image.LANCZOS)
    thumb.save(f"{out_dir}/{recipe_id}_thumb.webp",
               "WEBP", quality=Config.WEBP_QUALITY_THUMB)

    # 4) BLURHASH — progressif yükleme placeholder'ı
    small = img.resize((32, 24))
    bhash = blurhash.encode(small, Config.BLURHASH_X,
                            Config.BLURHASH_Y)

    return {
        "width": Config.FULL_WIDTH,
        "height": Config.FULL_HEIGHT,
        "blur_hash": bhash,
    }

def crop_to_ratio(img, rw, rh):
    """Görseli merkeze göre hedef orana kırp."""
    w, h = img.size
    target = rw / rh
    if w / h > target:          # çok geniş → yanları kes
        new_w = int(h * target)
        x = (w - new_w) // 2
        return img.crop((x, 0, x + new_w, h))
    else:                       # çok uzun → alt/üst kes
        new_h = int(w / target)
        y = (h - new_h) // 2
        return img.crop((0, y, w, y + new_h))

4.5 Ana Pipeline Döngüsü

Tüm parçaları birleştiren ana script. Kaynakları öncelik sırasıyla dener, standardize eder, sonucu kaydeder.

pipeline.py
import asyncio, aiohttp, json
from config import Config
from clients.unsplash_client import UnsplashClient
from clients.pexels_client import PexelsClient
from clients.pixabay_client import PixabayClient
from clients.ai_generator import AIGenerator
from processing.standardize import standardize

async def process_recipe(session, recipe, clients, ai_gen):
    """Tek tarif için görsel üret — kaynak zinciri."""
    query = recipe["search_query"]   # örn "lentil soup turkish"
    result = None

    # 1-3: Ücretsiz kaynakları sırayla dene
    for client in clients:               # [unsplash, pexels, pixabay]
        result = await client.search(session, query)
        if result and result["score"] >= Config.MATCH_SCORE_THRESHOLD:
            break
        result = None

    # 4: Hiçbiri yetmezse AI üret
    if result is None:
        result = await ai_gen.generate(
            recipe["title_en"], recipe["cuisine"])

    # Görseli indir
    async with session.get(result["url"]) as resp:
        raw = await resp.read()

    # Standardize et (1200x900 WebP + thumb + blur)
    meta = standardize(raw, recipe["id"], Config.OUTPUT_DIR)
    result.update(meta)
    result["recipe_id"] = recipe["id"]
    result["status"] = "ready"
    return result

async def main(recipes):
    """Tüm tarifleri paralel işle (5'erli grup)."""
    clients = [
        UnsplashClient(Config.UNSPLASH_KEY),
        PexelsClient(Config.PEXELS_KEY),
        PixabayClient(Config.PIXABAY_KEY),
    ]
    ai_gen = AIGenerator(Config.OPENAI_KEY)
    results = []

    async with aiohttp.ClientSession() as session:
        # 5 eşzamanlı — API limitlerini aşmadan
        sem = asyncio.Semaphore(5)
        async def guarded(r):
            async with sem:
                try:
                    return await process_recipe(
                        session, r, clients, ai_gen)
                except Exception as e:
                    return {"recipe_id": r["id"],
                            "status": "failed", "error": str(e)}
        results = await asyncio.gather(
            *[guarded(r) for r in recipes])

    # Sonuçları kaydet
    with open("results.json", "w") as f:
        json.dump(results, f, ensure_ascii=False, indent=2)
    ok = sum(1 for r in results if r["status"] == "ready")
    print(f"Tamamlandı: {ok}/{len(results)} başarılı")

if __name__ == "__main__":
    with open("recipes.json") as f:
        recipes = json.load(f)
    asyncio.run(main(recipes))
⚡ Paralel İşleme
Semaphore(5) ile aynı anda en fazla 5 tarif işlenir. Bu, API oran limitlerini aşmadan ~2500 görseli birkaç saatte tamamlar — tek iş parçacıklı yaklaşıma göre ~5 kat hızlı.

5.2 İş Kuyruğu ve Devam Edilebilirlik

Her tarif için bir iş kaydı tutulur (image_jobs.json). Durum alanları: pending, processing, completed, failed. Pipeline kesilse bile pending işlerden devam eder — tamamlanmış işler tekrar işlenmez.

📌 Özet

Bu pipeline, 2500 tariflik görsel içeriği $40–60 maliyetle ve birkaç saatte üretir. Hibrit kaynak stratejisi (ücretsiz stok + AI) maliyeti düşük tutar; Style Bible kuralları tüm görsellerin tutarlı olmasını sağlar; iş kuyruğu sistemi süreci kesintiye dayanıklı kılar.

Geliştiriciye düşen: API anahtarlarını edinmek, .env'i doldurmak, pilot grupla test etmek ve tam çalıştırmadır. AI görselleri için kısa bir insan inceleme adımı önerilir.
